Race Timer Leaderboard


public Text timerText;
private float startTime;
private bool isRaceStarted;
private static RaceTimer instance;

private void Start() {
    if (instance != null && instance != this) {
        Destroy(gameObject);
        return;
    }
    instance = this;
    StartRace();
}

private void Update() {
    if (isRaceStarted) {
        float currentTime = Time.time - startTime;
        int minutes = (int)(currentTime / 60);
        int seconds = (int)(currentTime % 60);
        int milliseconds = (int)((currentTime * 1000) % 1000);

        timerText.text = string.Format("{0:00}:{1:00}:{2:000}", minutes, seconds, milliseconds);
    }
}

public void StartRace() {
    startTime = Time.time;
    isRaceStarted = true;
}

public void EndRace() {
    float raceTime = Time.time - startTime;
    PlayerPrefs.SetFloat("RaceTime", raceTime);
}

private void OnDisable() {
    EndRace();
}

Car Controller Code


public class CarController : MonoBehaviour {

    [SerializeField] WheelCollider frontRight;
    [SerializeField] WheelCollider frontLeft;
    [SerializeField] WheelCollider backRight;
    [SerializeField] WheelCollider backLeft;

    [SerializeField] private TMP_Text speedText;
    [SerializeField] private Rigidbody rb;
    [SerializeField] private CountDownTimer countDownTimer;

    [SerializeField] private string inputNameHorizontal;
    [SerializeField] private string inputNameVertical;
    [SerializeField] private string inputBrake;

    public FinishlineChecker finishLineChecker;

    public float acceleration = 500f;
    public float brakingForce = 300f;
    public float maxTurnAngle = 15f;
    public float DownForceValue = 5000f;

    private float currentAcceleration = 0f;
    private float currentBrakeForce = 0f;
    private float currentTurnAngle = 0f;

    private void Update() {
        speedText.text = (rb.velocity.magnitude * 2.23693629f).ToString("0") + " kmph";
    }

    private void FixedUpdate() {
        if (countDownTimer != null && countDownTimer.currentTime <= 0) {
            currentAcceleration = acceleration * Input.GetAxis(inputNameVertical);
            currentBrakeForce = Input.GetButton(inputBrake) ? brakingForce : 0;

            ApplyAcceleration();
            ApplyBrakes();

            currentTurnAngle = maxTurnAngle * Input.GetAxis(inputNameHorizontal);
            frontLeft.steerAngle = currentTurnAngle;
            frontRight.steerAngle = currentTurnAngle;

            AddDownForce();
        }
    }

    private void ApplyAcceleration() {
        frontRight.motorTorque = currentAcceleration;
        frontLeft.motorTorque = currentAcceleration;
    }

    private void ApplyBrakes() {
        frontRight.brakeTorque = currentBrakeForce;
        frontLeft.brakeTorque = currentBrakeForce;
        backRight.brakeTorque = currentBrakeForce;
        backLeft.brakeTorque = currentBrakeForce;
    }

    private void AddDownForce() {
        rb.AddForce(-transform.up * DownForceValue * rb.velocity.magnitude);
    }

    private void OnTriggerEnter(Collider other) {
        if (other.CompareTag("Finish")) {
            string carName = gameObject.name;
            float finishTime = Time.time;
            finishLineChecker.ParticipantCrossedFinishLine(carName, finishTime);
        }
    }
}

public void Shoot(EntityType entityType) {
    hasPicked = true;
    Entity target = GameManager.Instance.GetEntity(entityType);
    if (target != null) {
        target.animator.SetTrigger("Pressed");
        if (entityType == this.entityType) target.animator.SetBool("Is Self", true);
        else animator.SetTrigger("Aim At Target");
        ShouldTalk();
    }
}

public void HitTarget(EntityType entityType) {
    if (animator.GetBool("Is Self") && animator.GetBool("Pressed")) {
        HitAnimation(this.entityType);
    } else if (GameManager.Instance.GetEntity(entityType).animator.GetCurrentAnimatorStateInfo(0).IsName("Pressed")) {
        HitAnimation(entityType);
    }
}

public void HitAnimation(EntityType entityType) {
    Entity target = GameManager.Instance.GetEntity(entityType);
    if (target != null) {
        bool isBullet = GameManager.Instance.Shoot();
        if (isBullet) {
            target.animator.SetTrigger("Hit");
        } else {
            GameManager.Instance.keepTurn = target.animator.GetBool("Is Self");
            AudioManager.Instance.PlaySoundOnce(AudioManager.Sound.DryFire); 
            Reset();
        }
    }
}

Explosion Logic in Unity (C#)

This script handles explosion logic when an object collides with a specified target.


                                    public AudioSource Audio;
                                    public TextMeshProUGUI score;
                                    public GameObject explosionPrefab;
                                    public string targetTag = "Ground"; // Set this to the tag you want to check
                                    private bool hasExploded = false;  // Prevent multiple explosions
                                    
                                    private void Start()
                                    {
                                        // Find the TextMeshProUGUI component in the scene
                                        score = GameObject.Find("ScoreText").GetComponent<TextMeshProUGUI>();
                                    }
                                    
                                    private void OnCollisionEnter(Collision collision)
                                    {
                                        // Check if the object that touched the cube is tagged as "Player"
                                        if (!hasExploded && collision.collider.CompareTag(targetTag))
                                        {
                                            // Find all objects with the tag "Enemy"
                                            GameObject[] enemies = GameObject.FindGameObjectsWithTag("Enemy");
                                            // Mark as exploded
                                            hasExploded = true;
                                    
                                            // Trigger the explosion at the collision point
                                            TriggerExplosion(transform.position);
                                    
                                            // Destroy each enemy object and update the score
                                            foreach (GameObject enemy in enemies)
                                            {
                                                Ghost ghostBehavior = enemy.GetComponent<Ghost>();
                                                if (ghostBehavior != null)
                                                {
                                                    ghostBehavior.GhostDeath();
                                                    ScoreManager.Instance.AddScore(1);
                                                }
                                            }
                                    
                                            // Update the score text after all enemies are destroyed
                                            score.text = "Score: " + ScoreManager.Instance.Score.ToString();
                                    
                                            Destroy(gameObject, 1f);
                                            Audio.Play();
                                        }
                                    }
                                    
                                    public void TriggerExplosion(Vector3 position)
                                    {
                                        // Instantiate the explosion prefab at the given position
                                        Instantiate(explosionPrefab, position, Quaternion.identity);
                                    }

                                        private void Update()
                                        {
                                            float distanceToPlayer = Vector3.Distance(transform.position, Player.position);
                                        
                                            if (State != GhostState.Death)
                                            {
                                                if (distanceToPlayer <= followDistance)
                                                {
                                                    State = GhostState.followplayer;
                                                }
                                                else if (State != GhostState.wander)
                                                {
                                                    State = GhostState.wander;
                                                }
                                            }
                                        
                                            switch (State)
                                            {
                                                case GhostState.idle:
                                                    GhostIdleState();
                                                    break;
                                                case GhostState.wander:
                                                    GhostWanderState();
                                                    break;
                                                case GhostState.followplayer:
                                                    GhostFollowPlayerState();
                                                    break;
                                                case GhostState.Death:
                                                    GhostDeath();
                                                    break;
                                            }
                                        }
                                        
                                        private IEnumerator HandleSpawnIdle()
                                        {
                                            State = GhostState.idle; // Begin in idle state
                                            yield return new WaitForSeconds(spawnIdleTime); // Wait for spawn idle time
                                            State = GhostState.wander; // Switch to wander state
                                        }
                                        
                                        public void SetReferences(Transform player, TextMeshProUGUI scoreUI)
                                        {
                                            Player = player;
                                            score = scoreUI;
                                        }
                                        
                                        private void GhostIdleState()
                                        {
                                            Agent.isStopped = true;
                                        }
                                        
                                        private void GhostWanderState()
                                        {
                                            Agent.isStopped = false;
                                        
                                            Timer -= Time.deltaTime;
                                            if (Timer <= 0)
                                            {
                                                MoveToRandomPosition();
                                                Timer = changeTime;
                                            }
                                        }
                                        
                                        private void GhostFollowPlayerState()
                                        {
                                            Agent.isStopped = false;
                                            Agent.SetDestination(Player.position);
                                        
                                            Vector3 directionToPlayer = (Player.position - transform.position).normalized;
                                            transform.LookAt(new Vector3(Player.position.x, transform.position.y, Player.position.z));
                                        }
                                        
                                        public void GhostDeath()
                                        {
                                            if (Agent != null)
                                            {
                                                Agent.isStopped = true;
                                                Debug.Log("agent off");
                                            }
                                            else
                                            {
                                                Debug.Log("no agent");
                                            }
                                        
                                            if (Animator != null)
                                            {
                                                Animator.SetTrigger("Dissolve"); 
                                            }
                                        
                                            Destroy(gameObject, 1f);
                                        }
                                        
                                        private void MoveToRandomPosition()
                                        {
                                            Vector3 randomDirection = Random.insideUnitSphere * moveRange;
                                            randomDirection += transform.position;
                                        
                                            NavMeshHit hit;
                                            if (NavMesh.SamplePosition(randomDirection, out hit, moveRange, NavMesh.AllAreas))
                                            {
                                                Agent.SetDestination(hit.position);
                                            }
                                        }
