Firerate Powerup

Met dit powerup kan je voor een bepaald tijd sneller schieten.

powerupC# 

using System.Collections;
using UnityEngine;

public class FireRatePower : MonoBehaviour {
    public AudioSource Audio;
    public Guns guns;
    private float originalFireRate;
    private MeshRenderer meshRenderer;
    private BoxCollider boxCollider;

    void Start() {
        guns = FindObjectOfType();
        if (guns != null) {
            originalFireRate = guns.fireRate;
        }
        meshRenderer = GetComponent();
        boxCollider = GetComponent();
    }

    private void OnTriggerEnter(Collider other) {
        if (other.CompareTag("powercolid") && guns != null) {
            StartCoroutine(ChangeFireRateAndDisable(0.1f, 5f));
            Audio.Play();
            Destroy(gameObject, 7f);
        }
    }

    private IEnumerator ChangeFireRateAndDisable(float newFireRate, float duration) {
        guns.fireRate = newFireRate;
        if (meshRenderer != null) meshRenderer.enabled = false;
        if (boxCollider != null) boxCollider.enabled = false;
        yield return new WaitForSeconds(duration);
        guns.fireRate = originalFireRate;
    }
}
De nuke powerup

De nuke powerup zorgt ervoor dat alle ghosts die er op het moment zijn dood gaan.

C#powerup 

using TMPro;
using UnityEngine;

public class Nuke : MonoBehaviour {
    public AudioSource Audio;
    [SerializeField] TextMeshProUGUI score;

    private void OnTriggerEnter(Collider other) {
        if (other.CompareTag("powercolid")) {
            GameObject[] enemies = GameObject.FindGameObjectsWithTag("Enemy");
            foreach (GameObject enemy in enemies) {
                Ghost ghostBehavior = enemy.GetComponent();
                if (ghostBehavior != null) {
                    ghostBehavior.GhostDeath();
                }
            }
            Destroy(gameObject, 1f);
            Audio.Play();
        }
    }
}
Powerup Spawner

om de zoveel tijd krijg je een powerup om je te helpen, zoals health, een nuke en een sneller schieten

C#Spawner 

using UnityEngine;
using System.Collections;

public class PowerupSpawner : MonoBehaviour {
    public GameObject[] prefabsToSpawn;
    public Transform[] spawnPoints;
    public float spawnInterval = 5f;
    public bool spawnOnStart = true;
    private Coroutine spawnRoutine;

    void Start() {
        if (spawnOnStart) {
            StartSpawning();
        }
    }

    public void StartSpawning() {
        if (spawnRoutine == null) {
            spawnRoutine = StartCoroutine(SpawnCoroutine());
        }
    }

    public void StopSpawning() {
        if (spawnRoutine != null) {
            StopCoroutine(spawnRoutine);
            spawnRoutine = null;
        }
    }

    private IEnumerator SpawnCoroutine() {
        while (true) {
            SpawnRandomPrefab();
            yield return new WaitForSeconds(spawnInterval);
        }
    }

    private void SpawnRandomPrefab() {
        if (prefabsToSpawn.Length == 0 || spawnPoints.Length == 0) return;
        GameObject prefab = prefabsToSpawn[Random.Range(0, prefabsToSpawn.Length)];
        Transform spawnPoint = spawnPoints[Random.Range(0, spawnPoints.Length)];
        Instantiate(prefab, spawnPoint.position, spawnPoint.rotation);
    }
}
